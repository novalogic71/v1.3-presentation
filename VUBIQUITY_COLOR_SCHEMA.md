```css
.btn {
  display: inline-flex;
  align-items: center;
  justify-content: center;
  border-radius: 12px; /* rounded-xl */
  font-weight: 500;
  transition: all 200ms;
  padding: 12px 24px; /* px-6 py-3 */

.btn:hover {
  transform: scale(1.02);
  box-shadow: 0 20px 25px -5px rgba(0, 0, 0, 0.1);

}

.btn:disabled {

}
```
